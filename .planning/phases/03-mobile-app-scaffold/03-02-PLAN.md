```javascript
    '!**/node_modules/**',
  ],
};
```

Create jest.setup.js:
```javascript
import '@testing-library/jest-native/extend-expect';

// Mock expo-router to prevent errors when imported
jest.mock('expo-router', () => ({
  Tabs: ({ children }: any) => children,
  Stack: ({ children }: any) => children,
  router: {
    push: jest.fn(),
    replace: jest.fn(),
  },
}));

// Mock react-native-mmkv for Wave 0 tests
jest.mock('react-native-mmkv', () => ({
  MMKV: jest.fn().mockImplementation(() => ({
    getString: jest.fn(),
    set: jest.fn(),
    delete: jest.fn(),
  })),
}));
```

Add test script to package.json scripts:
```json
"test": "jest",
"test:watch": "jest --watch"
```

```typescript
    expect(mockStorage.setItem).toBeDefined();
    expect(mockStorage.removeItem).toBeDefined();
  });

  it('getItem returns string or null', () => {
    const mockStorage: StateStorage = {
      getItem: (name: string): string | null => 'test-value',
      setItem: (name: string, value: string): void => {},
      removeItem: (name: string): void => {},
    };

    const result = mockStorage.getItem('test-key');
    expect(typeof result === 'string' || result === null).toBe(true);
  });

  it('setItem accepts string name and value', () => {
    let storedValue = '';
    const mockStorage: StateStorage = {
      getItem: (name: string): string | null => null,
      setItem: (name: string, value: string): void => {
        storedValue = value;

      removeItem: (name: string): void => {},
```
